Given a number find the number of trailing zeroes in its factorial.

Input Format

A single integer - N

Output Format

Print a single integer which is the number of trailing zeroes.

Input Constraints

1 <= N <= 1000

Example:

SAMPLE INPUT 
10

SAMPLE OUTPUT 
2

Explanation
10! = 3628800 has 2 zeros in the end.

#include 

long factorial10(long n)
{
  long c;
  long result = 0;
 
  for (c = 5; c <= n; c = c + 5){
  	long temp = c;
    while(temp%5 == 0){
    	result++;
    	temp = temp/5;
    }
  }
 
  return result;
}

int main()
{
	long fac;
    scanf("%ld", &fac);
    
    
    printf("%ld", factorial10(fac));
    
    return 0;
}
